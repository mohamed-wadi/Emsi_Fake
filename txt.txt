function buildCalendar() {
  const week = WEEKS[((weekOffset % WEEKS.length) + WEEKS.length) % WEEKS.length];
  document.getElementById('calLabel').textContent = week.label;

  const grid = document.getElementById('calGrid');
  grid.innerHTML = '';

  // Corner
  const corner = document.createElement('div');
  corner.className = 'cal-corner';
  grid.appendChild(corner);

  // Day headers
  week.days.forEach(d => {
    const hdr = document.createElement('div');
    hdr.className = 'cal-day-hdr';
    const num = d.today
      ? `<div class="dd today">${d.d}</div>`
      : `<div class="dd">${d.d}</div>`;
    hdr.innerHTML = `<div class="dn">${d.n}</div>${num}`;
    grid.appendChild(hdr);
  });

  // Time rows
  const events = WEEK_EVENTS[0] || [];
  HOUR_VALS.forEach((h, hi) => {
    // Time label
    const tl = document.createElement('div');
    tl.className = 'cal-time';
    tl.textContent = HOURS[hi];
    grid.appendChild(tl);

    // 7 day cells
    for (let col = 1; col <= 7; col++) {
      const cell = document.createElement('div');
      cell.className = 'cal-cell';

      // Find events starting in this hour & column (only for default week)
      if (weekOffset === 0) {
        events.forEach(ev => {
          if (ev.col === col && ev.startH === h) {
            const durationMins = (ev.endH - ev.startH) * 60 + (ev.endM - ev.startM);
            const heightPx = (durationMins / 60) * SLOT_H;
            const topOffset = (ev.startM / 60) * SLOT_H;

            const evDiv = document.createElement('div');
            evDiv.className = `ev ${ev.color}`;
            evDiv.style.cssText = `height:${heightPx - 4}px; top:${topOffset}px;`;
            evDiv.innerHTML = `<div class="et">${ev.title}</div><div class="ed">${ev.detail}</div>`;
            cell.appendChild(evDiv);
          }
        });
      }
      grid.appendChild(cell);
    }
  });
}

function calMove(dir) {
  weekOffset += dir;
  buildCalendar();
}

}

/* ═══════════════════ NAVIGATION ═══════════════════ */
function showPage(id) {
  // hide all pages
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  // deactivate all nav items
  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));

  // show target page
  const page = document.getElementById('page-' + id);
  if (page) page.classList.add('active');

  // highlight correct nav item
  const navItems = document.querySelectorAll('.nav-item');
  navItems.forEach(item => {
    if (item.getAttribute('onclick') === `showPage('${id}')`) {
      item.classList.add('active');
    }
  });

  // on mobile, close sidebar
  if (window.innerWidth < 768) {
    document.getElementById('sidebar').classList.add('collapsed');
  }
}

function toggleSidebar() {
  document.getElementById('sidebar').classList.toggle('collapsed');
}

/* ═══════════════════ AUTH ═══════════════════ */
function doLogin() {
  document.getElementById('loginPage').style.display = 'none';
  const portal = document.getElementById('portal');
  portal.style.display = 'flex';

  // initialise all dynamic content
  buildCalendar();
  buildAbsTable('tbAbsCoursS2', ABS_COURS_S2);
  buildAbsTable('tbAbsCoursS1', ABS_COURS_S1);
  buildTranscript();

  // set default active nav item
  document.querySelector('.nav-item').classList.add('active');
}

function doLogout() {
  document.getElementById('portal').style.display = 'none';
  document.getElementById('loginPage').style.display = 'flex';
}
